Selecionando

  • por Marca
  • por ID
  • por Nome
  • por Classe
  • por Seletor

por Marca

Selecionar mais de um elemento:
getElementsByTagName() 

        
            // 0 é o primeiro paragrafo q aparecer e assim sucessivamente
            // Precisa dizer qual paragrafo acessar
            var p1 = window.document.getElementsByTagName('p')[0]
            window.document.write('Está escrito assim: ' + p1.innerText)

por ID

Selecionar pela Id getElementById() 

        
            var d = window.document.getElementById('id')
            d.style.background = 'green'
            d.innerText = 'Estou aguardando...'

por Nome

selecionar por nome getElementsByName() 

        
            var m = window.document.getElementsByName('name')[0]
            m.innerText = 'Hello'

por Classe

getElementsByClassName() 

        
            var z = window.document.getElementsByClassName('class')[0]
            z.style.color = 'red'

por Seletor

querySelector()
querySelectorAll() 

        
            // Id na tag
            var d = window.document.querySelector('div#msg')
            d.style.background = 'blue'

            // classe na tag
            var e = window.document.querySelector('p.aqui')
            e.style.background = 'red'
