Auto-atribuições

        
            var n = 3
            n = n + 4               // 7
            n = n - 5               // 2
            n = n * 4               // 8
            n = n / 2               // 4
            n = n ** 2              // 16
            n = n % 5               // 1
        
        Simplificando
        
            var n = 3
            n += 4               // 7
            n -= 5               // 2
            n *= 4               // 8
            n /= 2               // 4
            n **= 2              // 16
            n %= 5               // 1
