Atribuição

= recebe

        
            var a = 5 + 3           // 8
            var b = a % 5           // 3
            var c = 5 * b ** 2      // 45
            var d = 10 - a / 2      // 6
            var e = 6 * 2 / d       // 2
            var f = b % e + 4 / 2   // 3

Auto-atribuições

        
            var n = 3
            n = n + 4               // 7
            n = n - 5               // 2
            n = n * 4               // 8
            n = n / 2               // 4
            n = n ** 2              // 16
            n = n % 5               // 1
        
        Simplificando
        
            var n = 3
            n += 4               // 7
            n -= 5               // 2
            n *= 4               // 8
            n /= 2               // 4
            n **= 2              // 16
            n %= 5               // 1

Relacionais

  • > (maior)
  • < (menor)
  • >= (maior igual)
  • <= (menor igual)
  • == (igual)
  • != (diferente)

Identidade

5 == 5 -> true
5 == '5' -> true // ele compara o número, não o tipo dele
5 === '5' -> false // ele compara o número e o tipo
=== (idêntico)

5 != '5' -> false
5 !== '5' -> true
!== (desigual restrito)

Lógicos

  • ! (negação)
  • && (e)
  • || (ou)

Negação

!
True False
False True

Conjunção

1° && 2° =
True E True True
True E False False
False E True False
False E False False

Disjunção

1° || 2° =
True OU True True
True OU False True
False OU True True
False OU False False
